CÓDIGO 2

#include <iostream>

int main() {
    int codigo;
    double precoCompra, precoVenda, lucro, totalCompra = 0, totalVenda = 0;
    int contLucroMenor10 = 0, contLucro10a20 = 0, contLucroMaior20 = 0;

    std::cout << "Digite o codigo da mercadoria ou -1 para sair: ";
    std::cin >> codigo;

    while(codigo != -1) {
        std::cout << "Digite o preco de compra: ";
        std::cin >> precoCompra;
        std::cout << "Digite o preco de venda: ";
        std::cin >> precoVenda;

        lucro = (precoVenda - precoCompra) / precoCompra * 100;
        totalCompra += precoCompra;
        totalVenda += precoVenda;

        if(lucro < 10) {
            contLucroMenor10++;
        } else if(lucro <= 20) {
            contLucro10a20++;
        } else {
            contLucroMaior20++;
        }

        std::cout << "Digite o codigo da mercadoria ou -1 para sair: ";
        std::cin >> codigo;
    }

    std::cout << "Quantidade de mercadorias com lucro menor que 10%: " << contLucroMenor10 << "\n";
    std::cout << "Quantidade de mercadorias com lucro entre 10% e 20%: " << contLucro10a20 << "\n";
    std::cout << "Quantidade de mercadorias com lucro maior que 20%: " << contLucroMaior20 << "\n";
    std::cout << "Valor total de compra: " << totalCompra << "\n";
    std::cout << "Valor total de venda: " << totalVenda << "\n";
    std::cout << "Lucro total: " << totalVenda - totalCompra << "\n";

    return 0;
}
