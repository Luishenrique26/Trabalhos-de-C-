#include <stdio.h>
#include <stdlib.h>

typedef struct pessoa {
    char nome [30];
    char cpf [15];
    int idade;
} Pessoa;

void limpa_linha() {
    scanf("%*[^\n]");
    scanf("%*c");
}

/*--------------------------------------------------------------------------------/
void gravar_registros() {

    FILE *arq;
    Pessoa pessoa;
    int i, n_reg;

    printf("\nQuantidade de registros para incluir: ");
    scanf("%d", &n_reg);
    limpa_linha();

    arq = fopen("arquivo.bin", "a+b"); // Leitura e escrita, colocando o ponteiro do arquivo no final deste, criando o arquivo caso o mesmo não exista.

    if (arq == NULL) {
        perror("Erro");
        exit(1);
    }
    for (i = 0; i < n_reg; i++) {

        printf("\nDigite o nome da pessoa: ");
        scanf("%[^\n]", pessoa.nome);
        limpa_linha();

        printf("Digite o cpf: ");
        scanf("%s", pessoa.cpf);
        limpa_linha();

        printf("Digite a idade: ");
        scanf("%d", &pessoa.idade);
        limpa_linha();

        fwrite(&pessoa, sizeof(Pessoa), 1, arq);
    }
    fclose(arq);
}

/*--------------------------------------------------------------------------------/
void ler_registros() {

    FILE *arquivo;
    Pessoa pessoa;

    arquivo = fopen("arquivo.bin", "rb");

    if (arquivo == NULL) {
        printf("erro ao abrir o arquivo.\n");
        exit(1);
    }

    printf("\n ========== PESSOAS CADASTRADAS ==========\n\n");

    fread(&pessoa, sizeof(Pessoa), 1, arquivo);
    do {
        printf("\nNOME: %s\nCPF: %s\nIDADE: %d\n", pessoa.nome, pessoa.cpf, pessoa.idade);
        fread(&pessoa, sizeof(Pessoa), 1, arquivo);
    } while (!feof(arquivo)); // enquanto não chegar ao final do arquivo

    fclose(arquivo);
}

/*--------------------------------------------------------------------------------/
int main(void) {

    char opcao;

    do {
        gravar_registros();

        ler_registros();

        printf("\nDeseja continuar? [s/n]: ");
        scanf("%c", &opcao);
        limpa_linha();

    } while (opcao == 's');

    return 0;
}

    string descricao;
    int prioridade;

void insere(Tarefa tarefas[], int& t, int tamanho) {
    if(tamanho == t)
        cout << "\nAtencao! lista cheia\n";
    else {
        cout << "\nDigite a descricao da tarefa: ";
        cin.ignore();
        getline(cin, tarefas[t].descricao);
        cout << "Digite a prioridade da tarefa (1-5): ";
        cin >> tarefas[t].prioridade;
        t++;
    }
}

void exibe(Tarefa tarefas[], int t) {
    if(t == 0)
        cout << "\nAtencao lista vazia\n";
    else {
        cout << "\nRelacao das tarefas\n";
        for(int x = 0; x < t; x++) {
            cout << "\n" << x + 1 << " - " << tarefas[x].descricao << " (Prioridade: " << tarefas[x].prioridade << ")";
        }
    }
}

void elemento(Tarefa tarefas[], int t) {
    int posicao;
    if(t == 0) {
        cout << "\nAtencao! lista vazia\n";
        return;

    cout << "\nQual a posicao que deseja? ";

    posicao = abs(posicao) - 1;
    if(posicao >= t) {
        cout << "\nAtencao! nenhuma tarefa armazenada ou posicao inexistente\n";
    } else {
        cout << "\nTarefa na posicao: " << tarefas[posicao].descricao << " (Prioridade: " << tarefas[posicao].prioridade << ")\n";
    }
}

# Inicializa a pilha P e a fila F com 7 elementos cada
P = list(range(1, 8))  # Pilha [1, 2, 3, 4, 5, 6, 7]
F = list(range(1, 8))  # Fila [1, 2, 3, 4, 5, 6, 7]

# Executa o pseudocódigo
for _ in range(len(P)):
    # Operação empilha(P, desempilha(P))
    elemento_p = P.pop()  # Remove do topo da pilha
    P.append(elemento_p)   # Insere no topo da pilha novamente

    # Operação enfileira(F, desenfileira(F))
    elemento_f = F.pop(0)  # Remove do início da fila
    F.append(elemento_f)    # Insere no final da fila

# Exibe os estados finais
print("Estado final da Pilha P:", P)
print("Estado final da Fila F:", F)
