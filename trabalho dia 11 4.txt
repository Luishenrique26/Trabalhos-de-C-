#include <iostream>
#include <cstdlib>
using namespace std;
 
int main()
{
  for (int i=0;i<100;i++)
  {
  	cout<<(i)<<("\n");
  }	;
 
  for (int i=0;i<100;i=i+2)
  {
  	cout<<(i)<<("\n");
  }	;
 
  for (int i=1;i<100;i=i+2)
  {
  	cout<<(i)<<("\n");
  }	;
}
